### 工作中用到的技巧

###### 数字格式化
```js
  //常用的数字千分位
  "1234567890".replace(/\B(?=(\d{3})+(?!\d))/g,",")
  (12345679890).toLocaleString()

  //格式化的数，转为普通的数(去逗号)
  "12,345,679,890".replace(/,/g,'')

  //强制保留两位小数
  var num = 99;
  num.toFixed(2)

  //去除小数末尾的0(主要的思想是转为数字)
  //如果能保证数字合法，可以使用弱转化，如果不能保证数字合法使用parseFloat
  //比如  var num = '996.33600.326.990000'; +num NaN
  //var num = '996.33600.326.990000'; parseFloat(num) 996.336
  var num = '996.33600';
  +num
  parseFloat(num)

```
